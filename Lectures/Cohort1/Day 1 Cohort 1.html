def mysum(x):
    """
    Sums a list or an array
    
    Input:
    x : a list or np.array of numbers
    
    Output:
    A scalar, the sum
    """
    s = 0
    for i in range(len(x)):
        if type(x[i]) == str: # avoid strings
            continue
        s += x[i]
    # Make sure to make return a function
    # 
    return(s)
